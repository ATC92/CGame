///< Debug version Windows build/Debug
cmake -G"MinGW Makefiles" ../.. -DCMAKE_BUILD_TYPE=Debug
cmake --build build/Debug --target Project -j 8

///< Release version
cmake -G"MinGW Makefiles" -B build/Release -DCMAKE_BUILD_TYPE=Release
cmake --build build/Release --target Project
cmake --install build/Release
cpack --config build/Release/CPackConfig.cmake



///< Debug Version Unix
cmake -G"Unix Makefiles" ../.. -DCMAKE_BUILD_TYPE=Debug
cmake --build build/Debug --target Project -j 8

///< Release Install Unix
cmake -G"Unix Makefiles" ../.. -DCMAKE_BUILD_TYPE=Release
cmake --install . --config Release --prefix ../Package
cpack -C Release
